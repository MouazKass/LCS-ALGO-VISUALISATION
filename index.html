Longest Common Subsequence (LCS) Algorithm Visualization

def lcs_length(X: str, Y: str) -> int: n, m = len(X), len(Y) L = [[0] * (m + 1) for _ in range(n + 1)] for i in range(1, n + 1): for j in range(1, m + 1): if X[i - 1] == Y[j - 1]: L[i][j] = L[i - 1][j - 1] + 1 else: L[i][j] = max(L[i - 1][j], L[i][j - 1]) return L[n][m]
